<!DOCTYPE html>
<html>
    <body>
        <input type="text" id="data" />
        <button onclick="downloadData()">Download</button>

        <script>
            function downloadData() {
                var data = document.getElementById("data").value;
                data = data.split(" ");
                console.log(data);
                for (var i = 0; i < data.length; i++) {
                    // Download the specification file
                    var specLink = document.createElement('a');
                    specLink.href = "sp.pdf";
                    specLink.download = data[i] + "Spec.pdf";
                    specLink.click();

                    // Download the installation file
                    var installLink = document.createElement('a');
                    installLink.href = "ins.pdf";
                    installLink.download = data[i] + "Install.pdf";
                    installLink.click();
                }
            }
        </script>
    </body>
</html>
